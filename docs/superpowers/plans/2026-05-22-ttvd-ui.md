# TTVD UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Package TikTokDownloader as a Nix CLI + Flask web UI that lets users paste a TikTok URL, watch the downloaded video in a browser, save it to their computer, or move it to a server-side folder.

**Architecture:** TikTokDownloader's GitHub source is pulled in as a flake input and packaged as a `buildPythonPackage` derivation (`ttvd`), making its `src.*` modules installable. The Flask app (`ttvdserver`) takes `ttvd` as a `propagatedBuildInput` so it can `from src.config import ...` directly. A NixOS module wires the Flask service into nginx on ATS machines only, and registers it on the `/` landing page via `machines.base.webServices`.

**Tech Stack:** Python 3.13, Flask, TikTokDownloader (JoeanAmier/TikTokDownloader), Nix/NixOS, nginx, systemd

```python
"""TTVD Flask server — TikTok video downloader UI."""
import argparse
import asyncio
import os
import shutil
import uuid
from pathlib import Path
from threading import Lock

import flask

parser = argparse.ArgumentParser(description="TTVD web server")
parser.add_argument("--port", type=int, default=6060)
parser.add_argument("--subdomain", type=str, default="/ttvd")
args = parser.parse_args()

SUBDOMAIN = args.subdomain.rstrip("/")
TEMP_ROOT = Path("/tmp/ttvd")
TEMP_ROOT.mkdir(parents=True, exist_ok=True)

app = flask.Flask(__name__, static_url_path=SUBDOMAIN)
bp = flask.Blueprint("ttvd", __name__)

_sessions: dict[str, Path] = {}
_lock = Lock()


def _build_parameter(root: Path):
    """Construct a minimal Parameter for unauthenticated public-video download."""
    from src.tools import ColorfulConsole
    from src.config import Settings, Parameter
    from src.manager import Database, DownloadRecorder
    from src.module import Cookie
    from src.record import BaseLogger

    root.mkdir(parents=True, exist_ok=True)
    console = ColorfulConsole()
    settings = Settings(root, console)
    cfg = settings.read()
    cfg["root"] = str(root)
    cfg["download"] = True
    cfg["music"] = False
    cfg["folder_mode"] = False

    db = Database(root / "db.sqlite", console)
    recorder = DownloadRecorder(db, False, console)
    cookie = Cookie(settings, console)

    return Parameter(
        settings,
        cookie,
        logger=BaseLogger,
        console=console,
        **cfg,
        recorder=recorder,
    )


async def _download_tiktok(url: str, dest: Path) -> dict:
    """Full pipeline: URL → mp4 in dest. Returns metadata dict."""
    from src.link import Extractor as LinkExtractor
    from src.interface import Detail
    from src.extract import Extractor
    from src.downloader import Downloader

    params = _build_parameter(dest)

    link_extractor = LinkExtractor(params)
    ids = await link_extractor.run(url)
    if not ids:
        raise ValueError("Could not extract video ID from URL")
    video_id = ids[0]

    detail = Detail(params, detail_id=video_id)
    raw_data = await detail.run()
    if not raw_data:
        raise ValueError("No metadata returned for this URL")

    extractor = Extractor(params)
    processed = await extractor.run(raw_data, params.recorder, type_="detail")
    if not processed:
        raise ValueError("Could not process video metadata")

    downloader = Downloader(params)
    await downloader.run(processed, "detail")

    mp4_files = list(dest.glob("**/*.mp4"))
    if not mp4_files:
        raise ValueError("Download finished but no mp4 file found")

    return {
        "file": mp4_files[0].name,
        "path": str(mp4_files[0]),
        "title": processed[0].get("desc", mp4_files[0].stem),
    }


@bp.route("/")
def index():
    return flask.send_file(
        os.path.join(os.path.dirname(__file__), "templates", "index.html")
    )


@bp.route("/api/fetch", methods=["POST"])
def fetch():
    data = flask.request.get_json() or {}
    url = data.get("url", "").strip()
    if not url:
        return flask.jsonify({"error": "Missing URL"}), 400

    token = str(uuid.uuid4())
    dest = TEMP_ROOT / token
    try:
        result = asyncio.run(_download_tiktok(url, dest))
    except Exception as exc:
        shutil.rmtree(dest, ignore_errors=True)
        return flask.jsonify({"error": str(exc)}), 500

    with _lock:
        _sessions[token] = Path(result["path"])

    return flask.jsonify({
        "token": token,
        "title": result["title"],
        "filename": result["file"],
    })


@bp.route("/api/stream/<token>")
def stream(token: str):
    with _lock:
        path = _sessions.get(token)
    if not path or not path.exists():
        flask.abort(404)
    return flask.send_file(str(path), mimetype="video/mp4")


@bp.route("/api/client-download/<token>")
def client_download(token: str):
    with _lock:
        path = _sessions.pop(token, None)
    if not path or not path.exists():
        flask.abort(404)
    response = flask.send_file(
        str(path),
        mimetype="video/mp4",
        as_attachment=True,
        download_name=path.name,
    )

    @response.call_on_close
    def _cleanup():
        shutil.rmtree(path.parent, ignore_errors=True)

    return response


@bp.route("/api/save-to-server", methods=["POST"])
def save_to_server():
    data = flask.request.get_json() or {}
    token = data.get("token", "")
    dest_dir = data.get("path", "").strip()

    with _lock:
        src_path = _sessions.pop(token, None)
    if not src_path or not src_path.exists():
        return flask.jsonify({"error": "Session not found or expired"}), 404
    if not dest_dir or not os.path.isdir(dest_dir):
        return flask.jsonify({"error": "Invalid destination directory"}), 400

    dest_path = Path(dest_dir) / src_path.name
    shutil.move(str(src_path), str(dest_path))
    shutil.rmtree(src_path.parent, ignore_errors=True)
    return flask.jsonify({"success": True, "saved_to": str(dest_path)})


@bp.route("/api/list-dirs", methods=["POST"])
def list_dirs():
    data = flask.request.get_json() or {}
    path = os.path.normpath(data.get("path", "/"))
    try:
        entries = os.listdir(path)
        dirs = sorted(e for e in entries
                      if os.path.isdir(os.path.join(path, e)) and not e.startswith("."))
        hidden = sorted(e for e in entries
                        if os.path.isdir(os.path.join(path, e)) and e.startswith("."))
        parent = os.path.dirname(path) if path != "/" else None
        return flask.jsonify({"path": path, "parent": parent, "dirs": dirs + hidden})
    except PermissionError:
        return flask.jsonify({"error": "Permission denied"}), 403
    except FileNotFoundError:
        return flask.jsonify({"error": "Path not found"}), 404


app.register_blueprint(bp, url_prefix=SUBDOMAIN)


def run():
    app.run(host="127.0.0.1", port=args.port, debug=False)


if __name__ == "__main__":
    run()
```

    const BASE = document.location.pathname.replace(/\/$/, '').replace(/\/[^\/]*$/, '') || '/ttvd';
    let currentToken = null;
    let currentFilename = null;
    let pickerPath = '/';

    function setError(msg) {
      const el = document.getElementById('error-msg');
      el.textContent = msg;
      el.style.display = msg ? 'block' : 'none';
    }

    function setSpinner(on) {
      document.getElementById('spinner').classList.toggle('active', on);
    }

    function setFetchBtn(enabled) {
      document.getElementById('fetch-btn').disabled = !enabled;
    }

    async function fetchVideo() {
      const url = document.getElementById('url-input').value.trim();
      if (!url) return;

      setError('');
      setSpinner(true);
      setFetchBtn(false);
      document.getElementById('player-section').classList.remove('active');

      try {
        const resp = await fetch(`${BASE}/api/fetch`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ url }),
        });
        const data = await resp.json();

        if (!resp.ok) {
          setError(data.error || 'Download failed');
          return;
        }

        currentToken = data.token;
        currentFilename = data.filename;

        const player = document.getElementById('video-player');
        player.src = `${BASE}/api/stream/${currentToken}`;
        player.load();

        document.getElementById('video-title').textContent = data.title;
        document.getElementById('player-section').classList.add('active');
      } catch (err) {
        setError('Network error: ' + err.message);
      } finally {
        setSpinner(false);
        setFetchBtn(true);
      }
    }

    function clientDownload() {
      if (!currentToken) return;
      const a = document.createElement('a');
      a.href = `${BASE}/api/client-download/${currentToken}`;
      a.download = currentFilename || 'video.mp4';
      a.click();
      // Token is consumed — hide player
      document.getElementById('player-section').classList.remove('active');
      currentToken = null;
    }

    async function loadDirs(path) {
      const resp = await fetch(`${BASE}/api/list-dirs`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ path }),
      });
      const data = await resp.json();
      if (data.error) { alert(data.error); return; }

      pickerPath = data.path;
      document.getElementById('modal-path').textContent = pickerPath;

      const list = document.getElementById('dir-list');
      list.innerHTML = '';

      if (data.parent !== null) {
        const item = document.createElement('div');
        item.className = 'dir-item parent';
        item.innerHTML = '&#8679; ..';
        item.onclick = () => loadDirs(data.parent);
        list.appendChild(item);
      }

      for (const dir of data.dirs) {
        const item = document.createElement('div');
        item.className = 'dir-item';
        item.innerHTML = `&#128193; ${dir}`;
        const fullPath = pickerPath === '/' ? '/' + dir : pickerPath + '/' + dir;
        item.onclick = () => loadDirs(fullPath);
        list.appendChild(item);
      }
    }

    function openFolderPicker() {
      if (!currentToken) return;
      pickerPath = '/';
      document.getElementById('modal').classList.add('active');
      loadDirs('/');
    }

    function closeModal() {
      document.getElementById('modal').classList.remove('active');
    }

    async function saveToServer() {
      if (!currentToken) return;
      const btn = document.getElementById('save-here-btn');
      btn.disabled = true;

      try {
        const resp = await fetch(`${BASE}/api/save-to-server`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ token: currentToken, path: pickerPath }),
        });
        const data = await resp.json();
        if (data.success) {
          closeModal();
          document.getElementById('player-section').classList.remove('active');
          setError('');
          currentToken = null;
          alert('Saved to: ' + data.saved_to);
        } else {
          alert('Error: ' + (data.error || 'Unknown error'));
        }
      } finally {
        btn.disabled = false;
      }
    }

    document.getElementById('url-input').addEventListener('keydown', e => {
      if (e.key === 'Enter') fetchVideo();
    });

    document.getElementById('modal').addEventListener('click', e => {
      if (e.target === document.getElementById('modal')) closeModal();
    });
